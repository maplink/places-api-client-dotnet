h4. Por Raio

bc.. var licenseInfo = new LicenseInfo { Key = "<chave da sua licença>", Login = "<login da sua licença>"};
var radius = 3.0;
var latitude = -23.45;
var longitude = -43.56;
var placeSearcher = new PlaceSearcher();
var results = placeSearcher.ByRadius(licenseInfo, radius, latitude, longitude);

h4. Por Termo

bc.. var licenseInfo = new LicenseInfo { Key = "<chave da sua licença>", Login = "<login da sua licença>"};
var term = "termo";
var placeSearcher = new PlaceSearcher();
var results = placeSearcher.ByTerm(licenseInfo, term);
